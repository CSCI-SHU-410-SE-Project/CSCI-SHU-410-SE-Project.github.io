1//! Internals of Deskulpt persistence-related macros.
2
3use proc_macro::TokenStream;
4use quote::quote;
5use syn::{parse_macro_input, parse_str, Data, DeriveInput, Expr, Fields, Ident, Lit, Meta, Type};
6
7/// Token stream processor for the `#[derive(Persisted)]` macro.
8pub fn proc_derive_persisted(input: TokenStream) -> TokenStream {
9    let input = parse_macro_input!(input as DeriveInput);
10
11    let original_name = &input.ident;
12    let persisted_name = Ident::new(&format!("{}Persisted", original_name), original_name.span());
13
14    // Struct-level #[serde(...)] attributes
15    let struct_serde_attrs: Vec<_> = input
16        .attrs
17        .iter()
18        .filter(|attr| attr.path().is_ident("serde"))
19        .collect();
20
21    let fields = match &input.data {
22        Data::Struct(data_struct) => match &data_struct.fields {
23            Fields::Named(fields_named) => &fields_named.named,
24            _ => panic!("Persisted can only be derived for structs with named fields"),
25        },
26        _ => panic!("Persisted can only be derived for structs"),
27    };
28
29    let mut persisted_fields = Vec::new();
30    let mut from_assignments = Vec::new();
31
32    for field in fields {
33        let field_name = field.ident.as_ref().unwrap();
34        let field_type = &field.ty;
35
36        // Field-level #[serde(...)] attributes
37        let field_serde_attrs: Vec<_> = field
38            .attrs
39            .iter()
40            .filter(|attr| attr.path().is_ident("serde"))
41            .collect();
42
43        // Parse field-level #[persisted(...)] attributes
44        let mut persisted_type = None;
45        let mut default_fn = None;
46        for attr in &field.attrs {
47            if attr.path().is_ident("persisted") {
48                match &attr.meta {
49                    Meta::List(_) => {
50                        attr.parse_nested_meta(|meta| {
51                            if meta.path.is_ident("type") {
52                                let value = meta.value()?;
53                                if let Expr::Lit(expr_lit) = value.parse()? {
54                                    if let Lit::Str(lit_str) = expr_lit.lit {
55                                        persisted_type = Some(lit_str.value());
56                                    }
57                                }
58                                Ok(())
59                            } else if meta.path.is_ident("default") {
60                                let value = meta.value()?;
61                                if let Expr::Lit(expr_lit) = value.parse()? {
62                                    if let Lit::Str(lit_str) = expr_lit.lit {
63                                        default_fn = Some(lit_str.value());
64                                    }
65                                }
66                                Ok(())
67                            } else {
68                                Err(meta.error("Unsupported #[persisted(...)] attribute"))
69                            }
70                        })
71                        .expect("Failed to parse #[persisted(...)] attribute");
72                    },
73                    _ => panic!("Invalid #[persisted(...)] attribute format"),
74                }
75            }
76        }
77
78        // Override field type
79        let persisted_field_type = if let Some(ref persisted_type) = persisted_type {
80            parse_str::<Type>(persisted_type)
81                .unwrap_or_else(|_| panic!("Failed to parse persisted type: {persisted_type}"))
82        } else {
83            field_type.clone()
84        };
85
86        // Override #[serde(default)] attribute
87        let default_serde_attr = if let Some(default_fn) = default_fn {
88            quote! { #[serde(default = #default_fn)] }
89        } else {
90            quote! { #[serde(default)] }
91        };
92
93        persisted_fields.push(quote! {
94            #[doc = concat!("[`", stringify!(#original_name), "::", stringify!(#field_name), "`]")]
95            #(#field_serde_attrs)*
96            #default_serde_attr
97            #field_name: #persisted_field_type
98        });
99
100        // If field type is overridden, we require it to be convertible via
101        // FromPersisted, otherwise no conversion is needed
102        let from_assignment = if persisted_type.is_some() {
103            quote! { #field_name: crate::settings::persistence::FromPersisted::from_persisted(persisted.#field_name) }
104        } else {
105            quote! { #field_name: persisted.#field_name }
106        };
107        from_assignments.push(from_assignment);
108    }
109
110    let expanded = quote! {
111        #[doc = concat!("Persisted [`", stringify!(#original_name), "`].")]
112        #[doc = ""]
113        #[doc = "Generated by [`deskulpt_macros::Persisted`]."]
114        #[derive(Default, ::serde::Deserialize)]
115        #(#struct_serde_attrs)*
116        pub(super) struct #persisted_name {
117            #(#persisted_fields,)*
118        }
119
120        impl From<#persisted_name> for #original_name {
121            fn from(persisted: #persisted_name) -> Self {
122                Self {
123                    #(#from_assignments,)*
124                }
125            }
126        }
127
128        impl crate::settings::persistence::FromPersisted<#persisted_name> for #original_name {
129            fn from_persisted(value: #persisted_name) -> Self {
130                value.into()
131            }
132        }
133    };
134    TokenStream::from(expanded)
135}
