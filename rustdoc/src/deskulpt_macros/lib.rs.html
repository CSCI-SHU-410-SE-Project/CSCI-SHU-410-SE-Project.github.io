11/// Derive the persisted version of a struct.
12///
13/// For `MyStruct`, this will generated a deserializable `MyStructPersisted`
14/// struct with each field marked with `#[serde(default)]`. If a field needs a
15/// custom default, it can be specified via `#[persisted(default = "...")]`
16/// which has the same semantic as `#[serde(default = "...")]`.
17///
18/// By default each field will be persisted with its original type, but this can
19/// be overridden via `#[persisted(type = "...")]`. This is useful for nested
20/// persisted structs.
21///
22/// This macro will further implement `From<MyStructPersisted>` and
23/// `FromPersisted<MyStructPersisted>` for `MyStruct`. The former is so that a
24/// `MyStructPersisted` loaded from persistence can be easily converted to
25/// `MyStruct` for use in the application. The latter is helpful for generating
26/// automatic conversion implementations.
27#[proc_macro_derive(Persisted, attributes(persisted))]
28pub fn derive_persisted(input: TokenStream) -> TokenStream {
29    persistence::proc_derive_persisted(input)
30}
