11/// Derive the
12/// [`DeskulptEvent`](../deskulpt_core/events/trait.DeskulptEvent.html) trait
13/// for a struct.
14///
15/// The event name is automatically generated by stripping off the "Event"
16/// suffix (if present) and converting the remaining name to kebab-case. For
17/// instance, `MyCustomEvent` will have the name `"my-custom"`. This behavior
18/// can be overridden with `#[deskulpt_event(name = "custom-name")]`.
19#[proc_macro_derive(DeskulptEvent, attributes(deskulpt_event))]
20pub fn derive_deskulpt_event(input: TokenStream) -> TokenStream {
21    events::proc_derive_deskulpt_event(input)
22}

24/// Register Deskulpt events for automatic TypeScript binding generation.
25///
26/// This macro generates a test function named `export_bindings_deskulpt_events`
27/// that exports TypeScript bindings for the registered Deskulpt event types.
28/// The bindings will be exported to `events.ts` under the same directory as the
29/// bindings generated by ts-rs.
30///
31/// ### Usage
32///
33/// ```ignore
34/// #[derive(DeskulptEvent, ts_rs::TS)]
35/// #[ts(export, export_to = "types.ts")]
36/// struct MyFirstEvent;
37///
38/// #[derive(DeskulptEvent, ts_rs::TS)]
39/// #[ts(export, export_to = "types.ts")]
40/// struct MySecondEvent;
41///
42/// register_deskulpt_events![MyFirstEvent, MySecondEvent];
43/// ```
44#[proc_macro]
45pub fn register_deskulpt_events(input: TokenStream) -> TokenStream {
46    events::proc_register_deskulpt_events(input)
47}
