12/// Data for an event in the bindings template.
13#[derive(Serialize)]
14pub struct BindingsTemplateEvent {
15    /// The key in the generated `events` object.
16    pub key: String,
17    /// The name of the event.
18    pub name: String,
19    /// The type representation of the event.
20    pub ty: String,
21}
22
23impl BindingsTemplateEvent {
24    /// Create an instance from event name and type.
25    pub fn from(ts: &Typescript, tcl: &TypeCollection, name: &str, ty: &DataType) -> Result<Self> {
26        Ok(Self {
27            key: name.to_lower_camel_case(),
28            name: name.to_string(),
29            ty: datatype(ts, &FunctionResultVariant::Value(ty.clone()), tcl)?,
30        })
31    }
32}

53/// Data for a command in the bindings template.
54#[derive(Serialize)]
55pub struct BindingsTemplateCommand {
56    /// The key in the generated `commands` object.
57    pub key: String,
58    /// The name of the command.
59    pub name: String,
60    /// The command arguments.
61    pub args: Vec<BindingsTemplateCommandArg>,
62    /// The type representation of the return value.
63    pub ret_ty: String,
64    /// The docstring of the command.
65    pub doc: String,
66}
67
68impl BindingsTemplateCommand {
69    /// Create an instance from function information.
70    pub fn from(ts: &Typescript, tcl: &TypeCollection, function: &Function) -> Result<Self> {
71        Ok(Self {
72            key: function.name().to_lower_camel_case(),
73            name: function.name().to_string(),
74            args: function
75                .args()
76                .map(|(name, ty)| BindingsTemplateCommandArg::from(ts, tcl, name, ty))
77                .collect::<Result<Vec<_>>>()?,
78            ret_ty: match function.result() {
79                Some(FunctionResultVariant::Value(t)) => {
80                    datatype(ts, &FunctionResultVariant::Value(t.clone()), tcl)?
81                },
82                Some(FunctionResultVariant::Result(t, _)) => {
83                    datatype(ts, &FunctionResultVariant::Value(t.clone()), tcl)?
84                },
85                None => "void".to_string(),
86            },
87            doc: {
88                let mut builder = js_doc::Builder::default();
89                if let Some(d) = &function.deprecated() {
90                    builder.push_deprecated(d);
91                }
92                if !function.docs().is_empty() {
93                    builder.extend(function.docs().split("\n"));
94                }
95                builder.build()
96            },
97        })
98    }
99}

101/// Data for the bindings template.
102#[derive(Serialize)]
103pub struct BindingsTemplate {
104    /// All types in the bindings.
105    pub types: Vec<String>,
106    /// All events in the bindings.
107    pub events: Vec<BindingsTemplateEvent>,
108    /// All commands in the bindings.
109    pub commands: Vec<BindingsTemplateCommand>,
110}
111
112impl BindingsTemplate {
113    /// Create an instance from export context.
114    pub fn from(ts: &Typescript, cfg: &ExportContext) -> Result<Self> {
115        Ok(Self {
116            types: cfg
117                .type_map
118                .into_iter()
119                .map(|(_, ndt)| Ok(export_named_datatype(ts, ndt, &cfg.type_map)?))
120                .collect::<Result<Vec<_>>>()?,
121            events: cfg
122                .events
123                .iter()
124                .map(|(name, ty)| BindingsTemplateEvent::from(ts, &cfg.type_map, name, ty))
125                .collect::<Result<Vec<_>>>()?,
126            commands: cfg
127                .commands
128                .iter()
129                .map(|function| BindingsTemplateCommand::from(ts, &cfg.type_map, function))
130                .collect::<Result<Vec<_>>>()?,
131        })
132    }
133}
