17/// Exporter for Deskulpt TypeScript bindings.
18#[derive(Default)]
19pub struct TypeScript(specta_typescript::Typescript);
20
21impl LanguageExt for TypeScript {
22    type Error = error::Error;
23
24    fn render(&self, cfg: &ExportContext) -> Result<String, Self::Error> {
25        let mut handlebars = Handlebars::new();
26        handlebars.register_escape_fn(handlebars::no_escape);
27        handlebars.register_helper("indent", Box::new(helpers::indent));
28        handlebars.register_template_string("bindings", include_str!("bindings.ts.hbs"))?;
29
30        let data = template::BindingsTemplate::from(&self.0, cfg)?;
31        let rendered = handlebars.render("bindings", &data)?;
32
33        // TODO: Remove when specta > 2.0.0-rc.22
34        let re =
35            Regex::new(r"Partial\s*<\s*(\{\s*\[\s*key\s+in\s+string\s*\][^}]*\})\s*>").unwrap();
36        let rendered = re.replace_all(&rendered, "$1").to_string();
37        Ok(rendered)
38    }
39
40    fn format(&self, path: &Path) -> Result<(), Self::Error> {
41        if let Some(formatter) = self.0.formatter {
42            formatter(path)?;
43        }
44        Ok(())
45    }
46}
