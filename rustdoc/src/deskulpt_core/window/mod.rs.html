13/// Deskulpt window enum.
14#[derive(specta::Type)]
15#[specta(rename_all = "camelCase")]
16pub enum DeskulptWindow {
17    /// The manager window.
18    Manager,
19    /// The canvas window.
20    Canvas,
21}
22
23impl DeskulptWindow {
24    /// Get the label of the window.
25    pub fn label(&self) -> &str {
26        match self {
27            DeskulptWindow::Manager => "manager",
28            DeskulptWindow::Canvas => "canvas",
29        }
30    }
31
32    /// Get the URL of the window.
33    pub fn url(&self) -> WebviewUrl {
34        match self {
35            DeskulptWindow::Manager => WebviewUrl::App("src/manager/index.html".into()),
36            DeskulptWindow::Canvas => WebviewUrl::App("src/canvas/index.html".into()),
37        }
38    }
39
40    /// Retrieve the webview window instance.
41    pub fn webview_window<R, M>(&self, manager: &M) -> Result<WebviewWindow<R>>
42    where
43        R: Runtime,
44        M: Manager<R> + ?Sized,
45    {
46        manager
47            .get_webview_window(self.label())
48            .ok_or_else(|| anyhow!("Window not found: {}", self.label()))
49    }
50}
51
52impl From<DeskulptWindow> for EventTarget {
53    fn from(window: DeskulptWindow) -> Self {
54        window.label().into()
55    }
56}

58/// Extention trait for window-related operations.
59pub trait WindowExt<R: Runtime>: Manager<R> + SettingsStateExt<R> {
60    /// Create the manager window.
61    fn create_manager(&self) -> Result<()>
62    where
63        Self: Sized,
64    {
65        let settings = self.get_settings();
66        let init_js = ManagerInitJS::generate(&settings)?;
67        WebviewWindowBuilder::new(
68            self,
69            DeskulptWindow::Manager.label(),
70            DeskulptWindow::Manager.url(),
71        )
72        .title("Deskulpt Manager")
73        .inner_size(800.0, 500.0)
74        .center()
75        .resizable(false)
76        .maximizable(false)
77        .minimizable(false)
78        .visible(false)
79        .initialization_script(&init_js)
80        .build()?;
81
82        Ok(())
83    }
84
85    /// Create the canvas window.
86    fn create_canvas(&self) -> Result<()>
87    where
88        Self: Sized,
89    {
90        let settings = self.get_settings();
91        let init_js = CanvasInitJS::generate(&settings)?;
92        let canvas = WebviewWindowBuilder::new(
93            self,
94            DeskulptWindow::Canvas.label(),
95            DeskulptWindow::Canvas.url(),
96        )
97        .maximized(true)
98        .transparent(true)
99        .decorations(false)
100        .always_on_bottom(true)
101        // TODO: Remove when the following issue is fixed:
102        // https://github.com/tauri-apps/tauri/issues/9597
103        .visible(false)
104        // Unsupported on macOS; see below for activation policy
105        .skip_taskbar(true)
106        .initialization_script(&init_js)
107        .build()?;
108
109        #[cfg(target_os = "macos")]
110        {
111            use objc2::msg_send;
112            use objc2::runtime::{AnyObject, Bool};
113
114            // Disable the window shadow on macOS; there will be shadows left on
115            // movement for transparent and undecorated windows that we are using;
116            // it seems that disabling shadows does not have significant visual impacts
117            unsafe {
118                let ns_window = canvas.ns_window()? as *mut AnyObject;
119                let () = msg_send![ns_window, setHasShadow:Bool::NO];
120            }
121        }
122
123        // TODO: Remove when the following issue is fixed:
124        // https://github.com/tauri-apps/tauri/issues/9597
125        canvas.show()?;
126
127        // Canvas is by default click-through
128        canvas.set_ignore_cursor_events(true)?;
129
130        Ok(())
131    }
132
133    /// Open the manager window.
134    fn open_manager(&self) -> Result<()> {
135        let manager = DeskulptWindow::Manager.webview_window(self)?;
136        manager.show()?;
137        manager.set_focus()?;
138        Ok(())
139    }
140}

145/// Window event handler for all Deskulpt windows.
146pub fn on_window_event(window: &Window, event: &WindowEvent) {
147    if window.label() == "manager" {
148        // Prevent the manager window from closing when the close button is
149        // clicked, but only hide it instead
150        if let WindowEvent::CloseRequested { api, .. } = event {
151            api.prevent_close();
152            if let Err(e) = window.hide() {
153                eprintln!("Failed to hide the manager window: {e}");
154            }
155        }
156    }
