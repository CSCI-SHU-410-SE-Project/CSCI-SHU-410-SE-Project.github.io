16/// Extention trait for system tray-related operations.
17pub trait TrayExt<R: Runtime>: StatesExtCanvasImode<R> {
18    /// Create the system tray.
19    fn create_tray(&self, icon: Image) -> Result<()>
20    where
21        Self: Sized,
22    {
23        // Store the menu item for toggling canvas interaction mode
24        let menu_item_toggle = MenuItemBuilder::with_id("tray-toggle", "Float").build(self)?;
25        self.set_canvas_imode_menu_item(&menu_item_toggle);
26
27        // Build the system tray menu
28        let tray_menu = MenuBuilder::new(self)
29            .items(&[
30                &menu_item_toggle,
31                &MenuItemBuilder::with_id("tray-manage", "Manage").build(self)?,
32                &MenuItemBuilder::with_id("tray-exit", "Exit").build(self)?,
33            ])
34            .build()?;
35
36        // Build the system tray icon
37        TrayIconBuilder::with_id("tray")
38            .icon(icon)
39            .icon_as_template(true)
40            .show_menu_on_left_click(false)
41            .tooltip("Deskulpt")
42            .menu(&tray_menu)
43            .on_menu_event(on_menu_event)
44            .on_tray_icon_event(on_tray_icon_event)
45            .build(self)?;
46
47        Ok(())
48    }
49}

51impl<R: Runtime> TrayExt<R> for App<R> {}
52impl<R: Runtime> TrayExt<R> for AppHandle<R> {}

54/// Handler for system tray menu events.
55///
56/// This handler will receive any menu event but only act on events related to
57/// the system tray.
58fn on_menu_event<R: Runtime>(app_handle: &AppHandle<R>, event: MenuEvent) {
59    match event.id().as_ref() {
60        "tray-toggle" => {
61            if let Err(e) = app_handle.toggle_canvas_imode() {
62                eprintln!("Error toggling canvas interaction mode: {e}");
63            }
64        },
65        "tray-manage" => {
66            if let Err(e) = app_handle.open_manager() {
67                eprintln!("Error opening manager window: {e}");
68            }
69        },
70        "tray-exit" => {
71            // Emit the "exit-app" event to the manager, which will invoke the
72            // `exit_app` command to clean up and actually exit the application
73            if let Err(e) = ExitAppEvent.emit_to(app_handle, DeskulptWindow::Manager) {
74                eprintln!("Failed to emit exit-app to manager: {e}");
75                app_handle.exit(1); // Safeguard exit
76            }
77
78            // Safeguard exit after 5 seconds if the normal exit procedure fails
79            let app_handle = app_handle.clone();
80            tauri::async_runtime::spawn(async move {
81                sleep(Duration::from_secs(5)).await;
82                app_handle.exit(1);
83            });
84        },
85        _ => {},
86    }
87}

89/// Handler for system tray icon events.
90fn on_tray_icon_event<R: Runtime>(tray: &TrayIcon<R>, event: TrayIconEvent) {
91    if let TrayIconEvent::Click {
92        button,
93        button_state,
94        ..
95    } = event
96    {
97        if button == MouseButton::Left && button_state == MouseButtonState::Down {
98            // Toggle canvas interaction mode on left-click
99            if let Err(e) = tray.app_handle().toggle_canvas_imode() {
100                eprintln!("Error toggling canvas interaction mode: {e}");
101            }
102        }
103    }
