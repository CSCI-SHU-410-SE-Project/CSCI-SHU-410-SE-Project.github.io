1//! State management for the initial render.
2
3use std::sync::Mutex;
4
5use anyhow::Result;
6use tauri::{App, AppHandle, Emitter, Manager, Runtime};
7use tauri_specta::Event;
8
9use crate::events::RenderWidgetsEvent;
10use crate::window::DeskulptWindow;
11
12/// Managed state for the initial render.
13///
14/// The first parameter indicates whether the canvas is ready to render widgets,
15/// i.e., whether the listener for the `render-widgets` event has been set up.
16/// The second parameter is an optional pending payload for that event. In
17/// particular, the manager window will that event to the canvas on startup, but
18/// the canvas may not be ready to receive the event at that point. In this
19/// case, we need to store the payload and emit it later when the canvas is
20/// ready.
21#[derive(Default)]
22struct InitialRenderState(Mutex<(bool, Option<RenderWidgetsEvent>)>);
23
24/// Extension trait for operations related to the initial render.
25pub trait StatesExtInitialRender<R: Runtime>: Manager<R> + Emitter<R> {
26    /// Initialize state management for the initial render.
27    fn manage_initial_render(&self) {
28        self.manage(InitialRenderState::default());
29    }
30
31    /// Set the canvas as ready to render widgets.
32    ///
33    /// If there is a pending payload, a `render-widgets` event will be emitted
34    /// to the canvas with that payload.
35    fn set_render_ready(&self) -> Result<()>
36    where
37        Self: Sized,
38    {
39        let state = self.state::<InitialRenderState>();
40        let mut render_ready = state.0.lock().unwrap();
41        render_ready.0 = true;
42
43        if let Some(event) = render_ready.1.take() {
44            event.emit_to(self, DeskulptWindow::Canvas)?;
45        }
46        Ok(())
47    }
48
49    /// Emit a `render-widgets` event to the canvas when it is ready.
50    ///
51    /// If the canvas is already ready to render widgets, emit the given payload
52    /// to the canvas immediately. Otherwise, store the payload as pending so
53    /// that it can be emitted later when the canvas is ready.
54    fn emit_on_render_ready(&self, event: RenderWidgetsEvent) -> Result<()>
55    where
56        Self: Sized,
57    {
58        let state = self.state::<InitialRenderState>();
59        let mut render_ready = state.0.lock().unwrap();
60
61        if !render_ready.0 {
62            render_ready.1 = Some(event);
63            return Ok(());
64        }
65        event.emit_to(self, DeskulptWindow::Canvas)?;
66        Ok(())
67    }
68}
69
70impl<R: Runtime> StatesExtInitialRender<R> for App<R> {}
71impl<R: Runtime> StatesExtInitialRender<R> for AppHandle<R> {}
